Quick Start

Three lines to get going.

// install
import { integrate, diff, plot } from 'math-project';

// compute a derivative
const result = diff('x^3 + 2*x', 'x'); // → "3x² + 2"

// plot it
plot('x^3 + 2*x', { x: [-10, 10] });
